  <script>

    const qs = [...document.querySelectorAll('.q')];

    const answers = { q1:null, q2:null, q3:null, q4:null, q5:null, q6:null };


    const staffEl = document.getElementById('staff');

    const deptEl = document.getElementById('dept');

    const reviewerEl = document.getElementById('reviewer');

    const roleEl = document.getElementById('role');

    const notesEl = document.getElementById('notes');


    const decisionBadge = document.getElementById('decisionBadge');

    const decisionTitle = document.getElementById('decisionTitle');

    const decisionReason = document.getElementById('decisionReason');

    const completeness = document.getElementById('completeness');


    const resetBtn = document.getElementById('resetBtn');

    const autoBtn = document.getElementById('autoBtn');

    const csvBtn = document.getElementById('csvBtn');

    const copyBtn = document.getElementById('copyBtn');


    // Role presets: pre-fill likely answers and hints

    const presets = {

      finance: { q1:'no', q2:'yes', q3:'yes', q4:'yes', q5:'sometimes', q6:'no' },

      me:      { q1:'no', q2:'no',  q3:'yes', q4:'yes', q5:'no',        q6:'no' },

      pm:      { q1:'no', q2:'no',  q3:'no',  q4:'yes', q5:'no',        q6:'no' },

      admin:   { q1:'no', q2:'no',  q3:'no',  q4:'no',  q5:'no',        q6:'yes' },

      field:   { q1:'no', q2:'no',  q3:'no',  q4:'no',  q5:'no',        q6:'yes' },

      it:      { q1:'no', q2:'yes', q3:'yes', q4:'yes', q5:'no',        q6:'no' },

    };


    function setAnswer(qIndex, val){

      // Map yes/no clicks. Ignore "sometimes" in UI; only in presets to nudge review.

      const key = 'q'+qIndex;

      answers[key] = (val === 'yes' ? 'yes' : 'no');

      // Update selected state

      const qEl = document.querySelector(`.q[data-q="${qIndex}"]`);

      qEl.querySelectorAll('.btn.yes, .btn.no').forEach(b=>b.classList.remove('selected'));

      qEl.querySelector(`.btn.${val}`).classList.add('selected');

      decide();

    }


    function clearSelections(){

      qs.forEach(q=>{

        q.querySelectorAll('.btn.yes, .btn.no').forEach(b=>b.classList.remove('selected'));

      });

    }


    function applyPreset(key){

      if(!key){return}

      const p = presets[key];

      // Fill answers

      Object.entries(p).forEach(([k,v])=>{

        const idx = Number(k.replace('q',''));

        const mapped = (v === 'yes' ? 'yes' : 'no'); // treat 'sometimes' as 'no' but we add note

        answers[k] = mapped;

        const qEl = document.querySelector(`.q[data-q="${idx}"]`);

        qEl.querySelectorAll('.btn.yes, .btn.no').forEach(b=>b.classList.remove('selected'));

        qEl.querySelector(`.btn.${mapped}`).classList.add('selected');

      });

      // Nudge for 'sometimes' in notes

      if (p.q5 === 'sometimes') {

        if (!notesEl.value.includes('Donor/compliance may apply; verify.')) {

          notesEl.value = (notesEl.value? notesEl.value.trim() + ' ' : '') + 'Donor/compliance may apply; verify.';

        }

      }

      decide();

    }


    // Decision logic per matrix:

    // - If any of Q1–Q5 is Yes, recommend Business (desktop). The first Yes provides rationale.

    // - Else if Q6 is Yes, recommend Basic/Web.

    // - Else if incomplete, show pending.

    function decide(){

      const completed = Object.values(answers).filter(v=>v==='yes' || v==='no').length;

      completeness.textContent = `${completed}/6 answered`;


      const triggers = [

        { key:'q1', label:'Requires offline or unreliable internet' },

        { key:'q2', label:'Needs desktop-only features or integrations' },

        { key:'q3', label:'Handles large files or performance-sensitive tasks' },

        { key:'q4', label:'Advanced editing beyond web capabilities' },

        { key:'q5', label:'Compliance/donor mandates desktop workflows or versions' },

      ];


      const firstYes = triggers.find(t => answers[t.key] === 'yes');


      if (firstYes){

        setDecision('Business (desktop)', 'danger',

          `Triggered by ${firstYes.label}. Default to Business; document justification and budget.`)

        return;

      }


      if (answers.q6 === 'yes' &&

          ['q1','q2','q3','q4','q5'].every(k => answers[k] !== 'yes')){

        setDecision('Basic/Web', 'ok',

          'No Business triggers (Q1–Q5) selected and usage is primarily light web-based.')

        return;

      }


      // If all answered but no clear signal (e.g., all No incl. Q6), suggest review

      if (completed === 6){

        setDecision('Review needed', 'warn',

          'No Business triggers and Q6 not confirmed. Verify role duties or run with Basic/Web by default.')

        return;

      }


      // Incomplete

      setDecision('No decision yet', 'warn', 'Answer all questions to receive a recommendation.');

    }


    function setDecision(title, style, reason){

      decisionTitle.textContent = title;

      decisionReason.textContent = reason;

      decisionBadge.className = 'badge ' + (style==='ok'?'ok':style==='danger'?'danger':'warn');

      decisionBadge.textContent = title;

    }


    function resetAll(){

      staffEl.value = '';

      deptEl.value = '';

      reviewerEl.value = '';

      roleEl.value = '';

      notesEl.value = '';

      Object.keys(answers).forEach(k=>answers[k]=null);

      clearSelections();

      decide();

    }


    function asCSVRow(){

      const now = new Date();

      const ts = now.toISOString();

      const decision = decisionTitle.textContent;

      const fields = [

        ['timestamp', ts],

        ['staff_name', staffEl.value],

        ['department', deptEl.value],

        ['role_preset', roleEl.value],

        ['reviewer', reviewerEl.value],

        ['q1_offline', answers.q1 ?? ''],

        ['q2_desktop_features', answers.q2 ?? ''],

        ['q3_large_files', answers.q3 ?? ''],

        ['q4_power_editing', answers.q4 ?? ''],

        ['q5_compliance', answers.q5 ?? ''],

        ['q6_light_web', answers.q6 ?? ''],

        ['decision', decision],

        ['rationale', notesEl.value.replace(/\n/g, ' ').trim()]

      ];


      const header = fields.map(([k])=>k).join(',');

      const row = fields.map(([,v])=>{

        const s = String(v ?? '');

        // Escape CSV

        if (s.includes('"') || s.includes(',') || s.includes('\n')){

          return '"' + s.replace(/"/g,'""') + '"';

        }

        return s;

      }).join(',');


      return { header, row };

    }


    function downloadCSV(){

      const { header, row } = asCSVRow();

      const csv = header + '\n' + row + '\n';

      const blob = new Blob([csv], { type:'text/csv;charset=utf-8;' });

      const a = document.createElement('a');

      a.href = URL.createObjectURL(blob);

      const staff = (staffEl.value || 'unknown').replace(/\s+/g,'_');

      a.download = `license-matrix_${staff}_${new Date().toISOString().slice(0,10)}.csv`;

      document.body.appendChild(a);

      a.click();

      a.remove();

    }


    function copySummary(){

      const { row } = asCSVRow();

      const decision = decisionTitle.textContent;

      const reason = decisionReason.textContent;

      const summary =

`License Matrix decision

- Staff: ${staffEl.value || '-'}

- Department/Project: ${deptEl.value || '-'}

- Role preset: ${roleEl.value || '-'}

- Reviewer: ${reviewerEl.value || '-'}

- Answers: Q1=${answers.q1||'-'}, Q2=${answers.q2||'-'}, Q3=${answers.q3||'-'}, Q4=${answers.q4||'-'}, Q5=${answers.q5||'-'}, Q6=${answers.q6||'-'}

- Decision: ${decision}

- Rationale: ${notesEl.value ? notesEl.value.trim() : reason}

- Timestamp: ${new Date().toLocaleString()}


CSV row:

${row}`;

      navigator.clipboard.writeText(summary).then(()=>{

        copyBtn.textContent = 'Copied!';

        setTimeout(()=>copyBtn.textContent='Copy summary', 1200);

      }).catch(()=>{

        alert('Unable to copy to clipboard. Your browser may block it.');

      });

    }


    // Event wiring

    qs.forEach(q=>{

      const idx = q.getAttribute('data-q');

      q.querySelectorAll('.btn.yes, .btn.no').forEach(btn=>{

        btn.addEventListener('click', ()=> setAnswer(idx, btn.classList.contains('yes') ? 'yes' : 'no'));

      });

    });
